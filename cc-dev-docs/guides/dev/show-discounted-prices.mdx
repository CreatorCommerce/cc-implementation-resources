## Overview

When a customer arrives via a creator's link, you can show them prices reflecting their exclusive discount. This creates urgency and improves conversion.

**Key Locations:**
- Product pages
- Collection/grid pages
- Quick-view modals
- Cart drawer/page
- Mini-cart

---

## Getting Discount Data

### From Creator Metaobject

```liquid
{% liquid
  if metaobject
    assign cc_creator = metaobject
  else
    assign cc_handle = cart.attributes['cc-creator-handle']
    assign cc_creator = metaobjects.creator[cc_handle]
  endif
  
  assign has_discount = cc_creator.cc-collab-discount-amount != blank
  assign discount_percent = cc_creator.cc-collab-discount-amount | default: 0
  assign discount_code = cc_creator.cc-collab-discount-code
%}
```

### Discount Types

| Field | Type | Example |
|-------|------|---------|
| `cc-collab-discount-amount` | Percentage | `15` (means 15%) |
| `cc-collab-discount-code` | String | `SARAH15` |
| `cc-collab-discount-type` | String | `percentage` or `fixed` |

## Product Page Pricing

### Basic Price Display

```liquid
{% comment %} snippets/cc-product-price.liquid {% endcomment %}

{% liquid
  assign original_price = product.price
  assign show_discount = false
  assign discounted_price = original_price
  
  if cc_creator and cc_creator.cc-collab-discount-amount
    assign discount_multiplier = 100 | minus: cc_creator.cc-collab-discount-amount | divided_by: 100.0
    assign discounted_price = original_price | times: discount_multiplier
    assign show_discount = true
  endif
%}

<div class="cc-price {% if show_discount %}cc-price--discounted{% endif %}">
  {% if show_discount %}
    <span class="cc-price__original">{{ original_price | money }}</span>
    <span class="cc-price__sale">{{ discounted_price | money }}</span>
    <span class="cc-price__badge">
      {{ cc_creator.cc-collab-discount-amount }}% OFF
    </span>
  {% else %}
    <span class="cc-price__regular">{{ original_price | money }}</span>
  {% endif %}
</div>

<style>
  .cc-price--discounted .cc-price__original {
    text-decoration: line-through;
    color: #999;
    font-size: 0.9em;
  }
  
  .cc-price__sale {
    color: #e74c3c;
    font-weight: bold;
    font-size: 1.2em;
  }
  
  .cc-price__badge {
    background: #e74c3c;
    color: white;
    padding: 2px 8px;
    border-radius: 4px;
    font-size: 0.75em;
    margin-left: 8px;
  }
</style>
```

### Handling Compare-at Prices

When the product already has a sale price:

```liquid
{% liquid
  assign base_price = product.price
  assign compare_at = product.compare_at_price
  
  if cc_creator and cc_creator.cc-collab-discount-amount
    assign discount_multiplier = 100 | minus: cc_creator.cc-collab-discount-amount | divided_by: 100.0
    
    if compare_at > base_price
      # Product already on sale, apply discount to sale price
      assign discounted_price = base_price | times: discount_multiplier
      assign show_compare = compare_at
    else
      # No existing sale, apply discount to regular price
      assign discounted_price = base_price | times: discount_multiplier
      assign show_compare = base_price
    endif
    
    assign show_discount = true
  else
    assign discounted_price = base_price
    assign show_compare = compare_at
    assign show_discount = compare_at > base_price
  endif
%}

<div class="cc-price">
  {% if show_compare > discounted_price %}
    <span class="cc-price__compare">{{ show_compare | money }}</span>
  {% endif %}
  <span class="cc-price__current {% if show_discount %}cc-price__current--sale{% endif %}">
    {{ discounted_price | money }}
  </span>
</div>
```

## Auto-Apply Discount Code

### Via Cart Attributes

```liquid

    // Store discount code for checkout

      body: JSON.stringify({
        attributes: {
          'cc-discount-code': '{{ cc_creator.cc-collab-discount-code | escape }}'
        }
      })
    });
  </script>
{% endif %}
```

## JavaScript Price Updates

### Dynamic Price Calculation

```javascript
// For dynamic updates (variants, quantity changes)
class CCPriceCalculator {
  constructor(discountPercent) {
    this.discountPercent = discountPercent;
    this.discountMultiplier = (100 - discountPercent) / 100;
  }
  
  calculate(price) {
    if (!this.discountPercent) return price;
    return Math.round(price * this.discountMultiplier);
  }
  
  formatMoney(cents) {
    return (cents / 100).toLocaleString('en-US', {
      style: 'currency',
      currency: 'USD'
    });
  }
  
  updatePriceDisplay(element, originalPrice) {
    const discounted = this.calculate(originalPrice);
    
    if (this.discountPercent) {
      element.innerHTML = `
        <span class="original">${this.formatMoney(originalPrice)}</span>
        <span class="sale">${this.formatMoney(discounted)}</span>
      `;
    } else {
      element.innerHTML = this.formatMoney(originalPrice);
    }
  }
}

// Initialize from Liquid
const ccPriceCalc = new CCPriceCalculator(
  {{ cc_creator.cc-collab-discount-amount | default: 0 }}
);
```

### Variant Change Handler

```javascript
document.querySelectorAll('[data-variant-select]').forEach(select => {
  select.addEventListener('change', function() {
    const variantId = this.value;
    const variant = window.productVariants.find(v => v.id == variantId);
    
    if (variant) {
      const priceEl = document.querySelector('[data-product-price]');
      ccPriceCalc.updatePriceDisplay(priceEl, variant.price);
    }
  });
});
```

## Handling Edge Cases

### Product Already Discounted

```liquid
{% liquid
  assign product_discount = 0
  if product.compare_at_price > product.price
    assign product_discount = product.compare_at_price | minus: product.price | times: 100 | divided_by: product.compare_at_price
  endif
  
  assign cc_discount = cc_creator.cc-collab-discount-amount | default: 0
  
  # Show which discount is better
  if cc_discount > product_discount
    assign best_discount = cc_discount
    assign best_discount_label = cc_creator.cc-creator-first-name | append: "'s discount"
    assign final_price = product.compare_at_price | default: product.price | times: discount_multiplier
  else
    assign best_discount = product_discount
    assign best_discount_label = "Sale"
    assign final_price = product.price
  endif
%}
```

### Minimum Price Guarantee

```liquid
{% liquid
  assign discounted_price = original_price | times: discount_multiplier
  
  # Ensure price doesn't go below a minimum (e.g., $1)
  assign min_price = 100
  if discounted_price < min_price
    assign discounted_price = min_price
  endif
%}
```

### Excluded Products

```liquid
{% liquid
  assign excluded_tags = 'no-discount,clearance,bundle' | split: ','
  assign allow_discount = true
  
  for tag in product.tags
    if excluded_tags contains tag
      assign allow_discount = false
      break
    endif
  endfor
  
  if allow_discount and cc_creator.cc-collab-discount-amount
    # Apply discount
  else
    # Show regular price
  endif
%}
```
