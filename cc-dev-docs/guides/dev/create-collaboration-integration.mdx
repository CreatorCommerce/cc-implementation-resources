## Overview

CreatorCommerce integrates with your existing tech stack through:

- **Unified API** — RESTful endpoints for all CC data
- **Webhooks** — Real-time event notifications
- **Shopify Flow** — Native Shopify automation triggers
- **Third-party connectors** — Klaviyo, affiliate platforms, etc.

This guide covers building custom integrations using the CC API.

---

## Authentication

### Getting API Credentials

1. Navigate to CreatorCommerce admin → Settings → API
2. Generate a new API key
3. Store securely (never commit to version control)

### Request Authentication

```bash
curl -X GET "https://api.creatorcommerce.com/v1/collabs" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json"
```

### Environment Variables

```bash
# .env (never commit this file)
CC_API_KEY=your_api_key_here
CC_API_BASE_URL=https://api.creatorcommerce.com/v1
```

## Webhook Integration

### Available Events

| Event | Trigger |
|-------|---------|
| `collab.created` | New partnership created |
| `collab.updated` | Partnership details changed |
| `collab.activated` | Partner activated their page |
| `drop.created` | New product drop created |
| `drop.updated` | Drop products/content changed |
| `order.attributed` | Order attributed to partner |

### Webhook Payload

```json
{
  "event": "collab.created",
  "timestamp": "2024-01-15T10:30:00Z",
  "data": {
    "collabId": "collab_123",
    "creatorId": "creator_456",
    "channelDomain": "your-store.myshopify.com"
  }
}
```

### Webhook Handler Example

```javascript
// Node.js/Express webhook handler
const express = require('express');
const crypto = require('crypto');

const app = express();

app.post('/webhooks/creatorcommerce', express.json(), (req, res) => {
  // Verify webhook signature
  const signature = req.headers['x-cc-signature'];
  const payload = JSON.stringify(req.body);
  const expectedSig = crypto
    .createHmac('sha256', process.env.CC_WEBHOOK_SECRET)
    .update(payload)
    .digest('hex');
  
  if (signature !== expectedSig) {
    return res.status(401).send('Invalid signature');
  }
  
  // Handle event
  const { event, data } = req.body;
  
  switch (event) {
    case 'collab.created':
      // Add to CRM, send welcome email, etc.
      handleNewCollab(data);
      break;
    case 'order.attributed':
      // Update affiliate platform, calculate commission
      handleAttributedOrder(data);
      break;
    default:
      console.log(`Unhandled event: ${event}`);
  }
  
  res.status(200).send('OK');
});
```

## Integration Patterns

### Sync to CRM (HubSpot Example)

```javascript
async function syncCreatorToHubSpot(creator) {
  const hubspotContact = {
    properties: {
      email: creator.email,
      firstname: creator.firstName,
      lastname: creator.lastName,
      cc_creator_handle: creator.handle,
      cc_tier: creator.tier,
      cc_discount_code: creator.discountCode,
      lifecyclestage: 'customer'
    }
  };
  
  await hubspot.crm.contacts.basicApi.create({ properties: hubspotContact });
}

// Webhook handler
async function handleNewCollab(data) {
  const collab = await ccApi.getCollab(data.collabId);
  await syncCreatorToHubSpot(collab.creator);
}
```

### Sync to Affiliate Platform

```javascript
async function syncToImpact(collab) {
  // Create or update partner in Impact
  const response = await fetch('https://api.impact.com/partners', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.IMPACT_API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      externalId: collab.id,
      name: `${collab.creator.firstName} ${collab.creator.lastName}`,
      email: collab.creator.email,
      promoCode: collab.discountCode,
      commissionRate: collab.commissionRate
    })
  });
  
  return response.json();
}
```

### Bi-directional Sync

```javascript
// Sync CC → External System
async function syncFromCC() {
  const collabs = await ccApi.listCollabs();
  for (const collab of collabs) {
    await externalSystem.upsertPartner(collab);
  }
}

// Sync External System → CC
async function syncToCC() {
  const externalPartners = await externalSystem.listPartners();
  for (const partner of externalPartners) {
    await ccApi.updateCollab(partner.ccCollabId, {
      customFields: {
        externalStatus: partner.status,
        externalTier: partner.tier
      }
    });
  }
}

// Run bi-directional sync
async function fullSync() {
  await syncFromCC();
  await syncToCC();
}
```

## Error Handling

### API Error Responses

```json
{
  "error": {
    "code": "COLLAB_NOT_FOUND",
    "message": "Collab with ID collab_123 not found",
    "status": 404
  }
}
```

### Retry Logic

```javascript
async function apiRequestWithRetry(requestFn, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await requestFn();
    } catch (error) {
      if (error.status === 429) {
        // Rate limited - wait and retry
        const waitTime = Math.pow(2, attempt) * 1000;
        await sleep(waitTime);
        continue;
      }
      if (attempt === maxRetries) throw error;
    }
  }
}
```
