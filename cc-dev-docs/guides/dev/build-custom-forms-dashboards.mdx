## Overview

CreatorCommerce provides built-in dashboards, but you may need:

- **Custom application forms** with your branding
- **Specialized data collection** for your use case
- **Embedded dashboards** in your existing tools
- **White-label partner portals**

This guide covers building these interfaces.

### Form Handler (Node.js)

```javascript
const express = require('express');
const multer = require('multer');
const { uploadToCloudinary } = require('./utils/cloudinary');
const { createPartner } = require('./utils/creatorcommerce');

const router = express.Router();
const upload = multer({ storage: multer.memoryStorage() });

router.post('/api/partner-application', 
  upload.single('profilePhoto'),
  async (req, res) => {
    try {
      const {
        firstName,
        lastName,
        email,
        instagram,
        tiktok,
        followerCount,
        bio
      } = req.body;
      
      // Upload profile photo if provided
      let profilePhotoUrl = null;
      if (req.file) {
        profilePhotoUrl = await uploadToCloudinary(req.file.buffer);
      }
      
      // Create partner in CreatorCommerce
      const partner = await createPartner({
        firstName,
        lastName,
        email,
        handle: generateHandle(firstName, lastName),
        customFields: {
          instagram,
          tiktok,
          followerCount,
          bio,
          profilePhoto: profilePhotoUrl,
          applicationDate: new Date().toISOString(),
          status: 'pending'
        }
      });
      
      // Send to your CRM
      await syncToCRM(partner);
      
      // Send confirmation email
      await sendWelcomeEmail(email, firstName);
      
      res.json({ success: true, partnerId: partner.id });
    } catch (error) {
      console.error('Application error:', error);
      res.status(500).json({ error: 'Application failed' });
    }
  }
);

function generateHandle(firstName, lastName) {
  const base = `${firstName}-${lastName}`.toLowerCase()
    .replace(/[^a-z0-9-]/g, '')
    .substring(0, 30);
  const suffix = Math.random().toString(36).substring(2, 6);
  return `${base}-${suffix}`;
}
```

### React Dashboard Component

```tsx
import React, { useState, useEffect } from 'react';
import { useAuth } from './hooks/useAuth';
import { ccApi } from './utils/creatorcommerce';

interface DashboardData {
  metrics: {
    clicks: number;
    orders: number;
    revenue: number;
    conversionRate: number;
  };
  recentOrders: Order[];
  pendingEarnings: number;
}

export function PartnerDashboard() {
  const { partner } = useAuth();
  const [data, setData] = useState<DashboardData | null>(null);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    async function fetchData() {
      try {
        const [metrics, orders, earnings] = await Promise.all([
          ccApi.getPartnerMetrics(partner.id),
          ccApi.getPartnerOrders(partner.id, { limit: 10 }),
          ccApi.getPartnerEarnings(partner.id)
        ]);
        
        setData({ metrics, recentOrders: orders, pendingEarnings: earnings });
      } catch (error) {
        console.error('Dashboard fetch error:', error);
      } finally {
        setLoading(false);
      }
    }
    
    fetchData();
  }, [partner.id]);
  
  if (loading) return <LoadingSpinner />;
  if (!data) return <ErrorState />;
  
  return (
    <div className="dashboard">
      <header className="dashboard-header">
        <h1>Welcome back, {partner.firstName}!</h1>
        <a href={partner.storefrontUrl} target="_blank" className="view-page-btn">
          View My Page →
        </a>
      </header>
      
      <section className="metrics-grid">
        <MetricCard 
          title="Page Views" 
          value={data.metrics.clicks} 
          change="+12%" 
        />
        <MetricCard 
          title="Orders" 
          value={data.metrics.orders} 
          change="+8%" 
        />
        <MetricCard 
          title="Revenue" 
          value={`$${data.metrics.revenue.toFixed(2)}`} 
          change="+15%" 
        />
        <MetricCard 
          title="Conversion" 
          value={`${(data.metrics.conversionRate * 100).toFixed(1)}%`} 
          change="+2%" 
        />
      </section>
      
      <section className="earnings-banner">
        <h3>Pending Earnings</h3>
        <p className="earnings-amount">${data.pendingEarnings.toFixed(2)}</p>
        <p className="earnings-note">Next payout: 1st of the month</p>
      </section>
      
      <section className="recent-orders">
        <h3>Recent Orders</h3>
        <OrderTable orders={data.recentOrders} />
      </section>
    </div>
  );
}

function MetricCard({ title, value, change }) {
  const isPositive = change.startsWith('+');
  
  return (
    <div className="metric-card">
      <h4>{title}</h4>
      <p className="metric-value">{value}</p>
      <span className={`metric-change ${isPositive ? 'positive' : 'negative'}`}>
        {change}
      </span>
    </div>
  );
}
```

## Product Selection Interface

### Product Picker Component

```tsx
interface ProductPickerProps {
  partnerId: string;
  onSelectionChange: (products: Product[]) => void;
  maxProducts?: number;
}

export function ProductPicker({ partnerId, onSelectionChange, maxProducts = 10 }: ProductPickerProps) {
  const [products, setProducts] = useState<Product[]>([]);
  const [selected, setSelected] = useState<Set<string>>(new Set());
  const [search, setSearch] = useState('');
  
  useEffect(() => {
    async function fetchProducts() {
      const allProducts = await shopifyApi.getProducts();
      setProducts(allProducts);
    }
    fetchProducts();
  }, []);
  
  const filteredProducts = products.filter(p => 
    p.title.toLowerCase().includes(search.toLowerCase())
  );
  
  const toggleProduct = (productId: string) => {
    const newSelected = new Set(selected);
    
    if (newSelected.has(productId)) {
      newSelected.delete(productId);
    } else if (newSelected.size < maxProducts) {
      newSelected.add(productId);
    }
    
    setSelected(newSelected);
    
    const selectedProducts = products.filter(p => newSelected.has(p.id));
    onSelectionChange(selectedProducts);
  };
  
  return (
    <div className="product-picker">
      <input
        type="search"
        placeholder="Search products..."
        value={search}
        onChange={(e) => setSearch(e.target.value)}
        className="product-search"
      />
      
      <p className="selection-count">
        {selected.size} of {maxProducts} selected
      </p>
      
      <div className="product-grid">
        {filteredProducts.map(product => (
          <div 
            key={product.id}
            className={`product-item ${selected.has(product.id) ? 'selected' : ''}`}
            onClick={() => toggleProduct(product.id)}
          >
            <img src={product.image} alt={product.title} />
            <p>{product.title}</p>
            <span className="price">{product.price}</span>
            {selected.has(product.id) && <CheckIcon className="check" />}
          </div>
        ))}
      </div>
    </div>
  );
}
```

## Form Validation

### Client-Side Validation

```javascript
const validationRules = {
  firstName: {
    required: true,
    minLength: 2,
    maxLength: 50
  },
  email: {
    required: true,
    pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/
  },
  instagram: {
    pattern: /^@?[\w.]+$/,
    maxLength: 30
  },
  bio: {
    required: true,
    minLength: 50,
    maxLength: 500
  }
};

function validateField(name, value) {
  const rules = validationRules[name];
  if (!rules) return { valid: true };
  
  if (rules.required && !value) {
    return { valid: false, error: 'This field is required' };
  }
  
  if (rules.minLength && value.length < rules.minLength) {
    return { valid: false, error: `Minimum ${rules.minLength} characters` };
  }
  
  if (rules.maxLength && value.length > rules.maxLength) {
    return { valid: false, error: `Maximum ${rules.maxLength} characters` };
  }
  
  if (rules.pattern && !rules.pattern.test(value)) {
    return { valid: false, error: 'Invalid format' };
  }
  
  return { valid: true };
}
```
